function fmt(n) {
  if(!n&&n!==0) return '–';
  if(n>=1e9) return `Rp ${(n/1e9).toFixed(2)} M`;
  if(n>=1e6) return `Rp ${(n/1e6).toFixed(2)} Jt`;
  return `Rp ${n.toLocaleString('id-ID')}`;
}
function fmtS(n) {
  if(n>=1e9) return `${(n/1e9).toFixed(1)}bn`;
  if(n>=1e6) return `${(n/1e6).toFixed(0)}jt`;
  return `${n}`;
}

function getFiltered(divisi) {
  return DATA.filter(r=>{
    if(r.divisi!==divisi) return false;
    if(state.selectedAM&&!state.selectedAM.includes(r.am)) return false;
    if(state.selectedStatus&&!state.selectedStatus.includes(r.phase)) return false;
    return true;
  });
}

function renderPanel(divisi) {
  const d=divisi.toLowerCase(), data=getFiltered(divisi);
  const total=data.reduce((s,r)=>s+r.nilai,0);
  const target=TARGETS[state.scope][divisi];
  const pct=target?(total/target*100):0;
  const shortage=total-target;
  const pctTxt=pct.toFixed(1)+'%';

  document.getElementById(`${d}-lop-label`).textContent=`${data.length} LOP`;
  document.getElementById(`${d}-real`).textContent=fmtS(total);
  document.getElementById(`${d}-target`).textContent=fmtS(target);
  document.getElementById(`${d}-target-lbl`).textContent=`Target ${state.scope}`;
  document.getElementById(`${d}-count`).textContent=data.length;
  document.getElementById(`${d}-pct`).textContent=pctTxt;
  document.getElementById(`${d}-gauge`).style.width=Math.min(pct,100)+'%';
  document.getElementById(`${d}-g-real`).textContent=fmtS(total);
  document.getElementById(`${d}-g-target`).textContent=fmtS(target);
  const sEl=document.getElementById(`${d}-shortage`);
  sEl.textContent=shortage>=0?'+'+fmtS(shortage):fmtS(shortage);
  sEl.className='shortage-val '+(shortage>=0?'pos':'neg');
  document.getElementById(`${d}-ach`).textContent=pctTxt;

  const maxC=Math.max(...PHASES.map(p=>data.filter(r=>r.phase===p).length),1);
  document.getElementById(`${d}-bars`).innerHTML=
    `<div class="barchart-title">LOP per Fase</div>`+
    PHASES.map(p=>{
      const rows=data.filter(r=>r.phase===p);
      const cnt=rows.length, val=rows.reduce((s,r)=>s+r.nilai,0);
      const w=cnt?(cnt/maxC*100).toFixed(0):0;
      return `<div class="bar-row">
        <span class="bar-lbl" style="color:${PHASE_COLORS[p]}">${p}</span>
        <div class="bar-track"><div class="bar-fill" style="background:${PHASE_COLORS[p]};width:${w}%"></div></div>
        <div class="bar-meta"><span class="bcount">${cnt}</span><span>${fmtS(val)}</span></div>
      </div>`;
    }).join('');
}

function renderTree(divisi) {
  const d=divisi.toLowerCase(), data=getFiltered(divisi);
  const ams=[...new Set(data.map(r=>r.am))].sort();
  const es=expandState[divisi];
  const rows=[];

  ams.forEach(am=>{
    const amData=data.filter(r=>r.am===am);
    const amVal=amData.reduce((s,r)=>s+r.nilai,0);
    const amOpen=!!es[am];
    const tag=divisi==='DPS'?`<span class="dtag tag-dps">DPS</span>`:`<span class="dtag tag-dss">DSS</span>`;
    rows.push(`<tr class="tr-am" onclick="toggleNode('${divisi}','${esc(am)}')">
      <td><div class="am-cell">
        <span class="chevron ${amOpen?'open':''}">▶</span>
        <span class="am-name">${am}</span>${tag}
        <span class="am-count">TOTAL ${amData.length} LOP</span>
      </div></td>
      <td class="td-meta"></td>
      <td class="td-right" style="color:var(--text2);font-size:11px">${fmt(amVal)}</td>
    </tr>`);

    if(!amOpen) return;

    PHASES.forEach(phase=>{
      const phData=amData.filter(r=>r.phase===phase);
      if(!phData.length) return;
      const phVal=phData.reduce((s,r)=>s+r.nilai,0);
      const key=`${am}|${phase}`;
      const phOpen=!!es[key];

      rows.push(`<tr class="tr-phase" onclick="toggleNode('${divisi}','${esc(key)}')">
        <td><div class="ph-cell">
          <span class="chevron ${phOpen?'open':''}">▶</span>
          <span class="ph-badge pb-${phase.toLowerCase()}">${phase}</span>
          <span class="ph-name">${PHASE_NAMES[phase]}</span>
          <span class="ph-cnt">${phData.length} proyek</span>
        </div></td>
        <td class="td-meta"></td>
        <td class="td-right" style="color:var(--text3);font-size:11px">${fmt(phVal)}</td>
      </tr>`);

      if(!phOpen) return;

      phData.forEach(r=>{
        rows.push(`<tr class="tr-detail">
          <td class="col-proyek">${r.proyek}</td>
          <td class="col-lopid">${r.lopid}</td>
          <td class="col-nilai">${fmt(r.nilai)}</td>
        </tr>`);
      });
    });
  });

  const grandVal=data.reduce((s,r)=>s+r.nilai,0);
  rows.push(`<tr class="tr-total">
    <td>Total</td>
    <td class="td-meta">${data.length} LOP</td>
    <td class="td-right">${fmt(grandVal)}</td>
  </tr>`);

  document.getElementById(`${d}-tree`).innerHTML=rows.join('');
}

}
